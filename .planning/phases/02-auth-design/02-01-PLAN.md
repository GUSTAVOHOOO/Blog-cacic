  <action>
  Create **src/components/providers/chakra-provider.tsx**:

  ```typescript
  'use client'
  import { ChakraProvider } from '@chakra-ui/react'
  import { ColorModeProvider } from '@chakra-ui/color-mode'
  import { system } from '@/lib/theme/config'
  import { ThemeProvider } from 'next-themes'

  export function Providers({ children }: { children: React.ReactNode }) {
    return (
      <ThemeProvider attribute="class" defaultTheme="light">
        <ChakraProvider value={system}>
          <ColorModeProvider>
            {children}
          </ColorModeProvider>
        </ChakraProvider>
      </ThemeProvider>
    )
  }
  ```

  Notes:
  - 'use client' directive at top (this is a client component)
  - ThemeProvider wraps everything to handle dark mode persistence
  - ChakraProvider applies the theme system
  - ColorModeProvider enables the color mode hook
  - Nesting order matters: ThemeProvider → ChakraProvider → ColorModeProvider

  Verify file is created and syntax is correct.
  </action>
