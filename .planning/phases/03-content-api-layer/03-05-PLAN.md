**src/app/api/contato/route.ts:**

Pattern: rate check → validate → save → respond.

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { contatoSchema } from '@/lib/validations/contato'
import { contatoRateLimiter } from '@/lib/rate-limit'
import { createContato } from '@/lib/services/contatos'

export const runtime = 'edge'  // Upstash rate limit works on Edge

export async function POST(request: NextRequest) {
  // 1. Rate limit by IP
  const ip = request.headers.get('x-forwarded-for')?.split(',')[0]?.trim()
    ?? request.headers.get('x-real-ip')
    ?? 'anonymous'

  const { success, remaining } = await contatoRateLimiter(ip)
  if (!success) {
    return NextResponse.json(
      { error: 'Muitas tentativas. Tente novamente em alguns minutos.' },
      { status: 429, headers: { 'X-RateLimit-Remaining': '0' } }
    )
  }

  // 2. Parse and validate body
  let body: unknown
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Corpo da requisição inválido' }, { status: 400 })
  }

  const result = contatoSchema.safeParse(body)
  if (!result.success) {
    return NextResponse.json(
      { error: 'Dados inválidos', details: result.error.flatten().fieldErrors },
      { status: 400 }
    )
  }

  // 3. Save to Supabase via DAL
  const saved = await createContato(result.data)
  if (!saved.success) {
    // SEC-08: generic error to client
    return NextResponse.json({ error: saved.error }, { status: 500 })
  }

  return NextResponse.json(
    { message: 'Mensagem enviada com sucesso!' },
    { status: 201 }
  )
}
```

import { NextRequest, NextResponse } from 'next/server'
import { recordView } from '@/lib/services/views'

export async function POST(request: NextRequest) {
  let body: { slug?: string }
  try {
    body = await request.json()
  } catch {
    return NextResponse.json({ error: 'Corpo inválido' }, { status: 400 })
  }

  if (!body.slug || typeof body.slug !== 'string' || body.slug.trim() === '') {
    return NextResponse.json({ error: 'slug é obrigatório' }, { status: 400 })
  }

  await recordView(body.slug.trim())
  return NextResponse.json({ success: true }, { status: 200 })
}

**src/app/api/upload/route.ts:**

Uses FormData (multipart), validates file, stores with UUID filename in Supabase Storage.

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { validateUpload } from '@/lib/validations/upload'
import { supabaseAdmin } from '@/lib/supabase/admin'
import { createClient } from '@/lib/supabase/server'

export async function POST(request: NextRequest) {
  // SEC-09: Require authentication for uploads
  const supabase = await createClient()
  const { data: { user }, error: authError } = await supabase.auth.getUser()
  if (authError || !user) {
    return NextResponse.json({ error: 'Autenticação necessária' }, { status: 401 })
  }

  // Parse multipart form
  let formData: FormData
  try {
    formData = await request.formData()
  } catch {
    return NextResponse.json({ error: 'Formato de requisição inválido' }, { status: 400 })
  }

  const file = formData.get('file') as File | null
  if (!file) {
    return NextResponse.json({ error: 'Arquivo não encontrado na requisição' }, { status: 400 })
  }

  // Validate type and size (SEC-06: never trust client-provided type)
  const validation = validateUpload({
    type: file.type,
    size: file.size,
    name: file.name,
  })

  if (!validation.valid) {
    return NextResponse.json({ error: validation.error }, { status: 400 })
  }

  // Generate UUID filename — never use original name (SEC-07)
  const ext = file.name.split('.').pop()?.toLowerCase() ?? ''
  const uuid = crypto.randomUUID()
  const filename = `${uuid}.${ext}`

  // Select storage bucket based on file category
  const bucket = validation.category === 'document' ? 'trabalhos-pdfs' : 'thumbnails'

  // Upload to Supabase Storage via admin client (bypasses RLS for storage)
  const arrayBuffer = await file.arrayBuffer()
  const { data: uploadData, error: uploadError } = await supabaseAdmin
    .storage
    .from(bucket)
    .upload(filename, arrayBuffer, {
      contentType: file.type,
      upsert: false,
    })

  if (uploadError) {
    // SEC-08: log details server-side, generic message to client
    console.error('[upload] Supabase storage error:', uploadError)
    return NextResponse.json({ error: 'Erro ao fazer upload do arquivo' }, { status: 500 })
  }

  // Return public URL
  const { data: { publicUrl } } = supabaseAdmin
    .storage
    .from(bucket)
    .getPublicUrl(filename)

  return NextResponse.json(
    { url: publicUrl, filename, bucket, category: validation.category },
    { status: 201 }
  )
}
```

**src/app/api/eventos/route.ts (GET public + POST auth):**

GET: calls `getEventos(options)` from services. Accepts query params `?tipo=palestra&limit=20`.
POST: SEC-09 — verify auth + check user role is 'editor' or 'admin' from profiles table before creating. Validate body with `createEventoSchema`. Return 201 with created evento.

```typescript
// GET — public
export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url)
  const tipo = searchParams.get('tipo') ?? undefined
  const limit = parseInt(searchParams.get('limit') ?? '20')

  try {
    const eventos = await getEventos({ tipo, limit })
    return NextResponse.json(eventos)
  } catch (err) {
    console.error('[api/eventos GET]', err)
    return NextResponse.json({ error: 'Erro interno' }, { status: 500 })
  }
}

// POST — editor/admin only
export async function POST(request: NextRequest) {
  // Auth check
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return NextResponse.json({ error: 'Não autorizado' }, { status: 401 })

  // Role check (SEC-09)
  const { data: profile } = await supabase
    .from('profiles')
    .select('role')
    .eq('id', user.id)
    .single()

  if (!profile || !['editor', 'admin'].includes(profile.role)) {
    return NextResponse.json({ error: 'Permissão insuficiente' }, { status: 403 })
  }

  // Validate + create
  const body = await request.json().catch(() => null)
  const result = createEventoSchema.safeParse(body)
  if (!result.success) {
    return NextResponse.json({ error: 'Dados inválidos', details: result.error.flatten().fieldErrors }, { status: 400 })
  }

  try {
    const evento = await createEvento(result.data)
    return NextResponse.json(evento, { status: 201 })
  } catch (err) {
    console.error('[api/eventos POST]', err)
    return NextResponse.json({ error: 'Erro ao criar evento' }, { status: 500 })
  }
}
```
